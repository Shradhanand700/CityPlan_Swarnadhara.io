class MaxHeap {
    int arr[100];
    int size;

public:
    MaxHeap() {
        size = 0;
    }

    // Insert element into heap
    void insert(int val) {
        size++;
        int i = size;

        // Move up until heap property is satisfied
        while (i > 1 && arr[i / 2] < val) {
            arr[i] = arr[i / 2];
            i = i / 2;
        }
        arr[i] = val;
    }

    // Delete maximum element (root)
    void deleteMax() {
        if (size == 0) {
            cout << "Heap is empty\n";
            return;
        }

        int last = arr[size];
        size--;

        int i = 1;
        int child;

        while (2 * i <= size) {
            child = 2 * i;

            if (child + 1 <= size && arr[child + 1] > arr[child])
                child++;

            if (last >= arr[child])
                break;

            arr[i] = arr[child];
            i = child;
        }
        arr[i] = last;
    }

    // Display heap
    void display() {
        for (int i = 1; i <= size; i++)
            cout << arr[i] << " ";
        cout << endl;
    }
};

int main() {
    MaxHeap h;

    h.insert(10);
    h.insert(40);
    h.insert(30);
    h.insert(60);
    h.insert(20);

    cout << "Heap elements: ";
    h.display();

    h.deleteMax();

    cout << "After deleting max: ";
    h.display();

    return 0;
}

void bfs(int m[10][10], int v, int source) {
    int queue[20];
    int front = 0, rear = 0, u, i;
    int visited[10];

    for (i = 0; i < v; i++)
        visited[i] = 0;

    queue[rear] = source;
    visited[source] = 1;

    cout << "The BFS Traversal is... \n";

    while (front <= rear) {
        u = queue[front];
        cout << u << "\t";
        front++;

        for (i = 0; i < v; i++) {
            if (m[u][i] == 1 && visited[i] == 0) {
                visited[i] = 1;
                rear++;
                queue[rear] = i;
            }
        }
    }
}

void bubbleSort(int A[], int n) {
    for (int i = 0; i <= n - 2; i++) {
        for (int j = 0; j <= n - 2 - i; j++) {
            if (A[j + 1] < A[j]) {
                swap(A[j], A[j + 1]);
            }
        }
    }
}

void selectionSort(int A[], int n) {
    for (int i = 0; i <= n - 2; i++) {
        int minIndex = i;
        for (int j = i + 1; j <= n - 1; j++) {
            if (A[j] < A[minIndex]) {
                minIndex = j;
            }
        }
        swap(A[i], A[minIndex]);
    }
}

void insertionSort(int A[], int n) {
    for (int i = 1; i <= n - 1; i++) {
        int v = A[i];
        int j = i - 1;

        while (j >= 0 && A[j] > v) {
            A[j + 1] = A[j];
            j--;
        }
        A[j + 1] = v;
    }
}

void merge(int A[], int left, int mid, int right) {
    int n1 = mid - left + 1;
    int n2 = right - mid;

    int L[n1], R[n2];

    for (int i = 0; i < n1; i++)
        L[i] = A[left + i];
    for (int j = 0; j < n2; j++)
        R[j] = A[mid + 1 + j];

    int i = 0, j = 0, k = left;

    while (i < n1 && j < n2) {
        if (L[i] <= R[j]) {
            A[k++] = L[i++];
        } else {
            A[k++] = R[j++];
        }
    }

    while (i < n1)
        A[k++] = L[i++];

    while (j < n2)
        A[k++] = R[j++];
}

void mergeSort(int A[], int left, int right) {
    if (left < right) {
        int mid = (left + right) / 2;

        mergeSort(A, left, mid);
        mergeSort(A, mid + 1, right);
        merge(A, left, mid, right);
    }
}

int partition(int A[], int l, int r) {
    int pivot = A[l];
    int i = l;
    int j = r + 1;

    do {
        do { i++; } while (A[i] < pivot);
        do { j--; } while (A[j] > pivot);

        if (i < j)
            swap(A[i], A[j]);
    } while (i < j);

    swap(A[l], A[j]);
    return j;
}

void quickSort(int A[], int l, int r) {
    if (l < r) {
        int s = partition(A, l, r);
        quickSort(A, l, s - 1);
        quickSort(A, s + 1, r);
    }
}

#define MAXN 100
#define INF 999

int main() {
    int n;
    cout << "Enter the number of Vertices\n";
    cin >> n;

    int cost[MAXN][MAXN];

    cout << "Enter the cost matrix: (Enter 999 for Infinity)\n";
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            cin >> cost[i][j];
        }
    }

    int source;
    cout << "Enter the source\n";
    cin >> source;

    int dist[MAXN], path[MAXN], visited[MAXN];


    for (int i = 0; i < n; i++) {
        dist[i] = cost[source][i];
        path[i] = source;
        visited[i] = 0;
    }
    visited[source] = 1;


    for (int iter = 0; iter < n - 1; iter++) {
        int u = -1;
        int min = INF + 1;


        for (int i = 0; i < n; i++) {
            if (!visited[i] && dist[i] < min) {
                min = dist[i];
                u = i;
            }
        }

        if (u == -1)
            break;

        visited[u] = 1;


        for (int v = 0; v < n; v++) {
            if (!visited[v] && cost[u][v] != INF) {
                if (dist[u] + cost[u][v] < dist[v]) {
                    dist[v] = dist[u] + cost[u][v];
                    path[v] = u;
                }
            }
        }
    }

    cout << "\nVertex   Dist   Path\n";
    for (int i = 0; i < n; i++) {
        cout << i << "        " << dist[i] << "      " << path[i] << "\n";
    }

    return 0;
}

    #include 
using namespace std;


struct Edge {
    int u, v, w;
};


void bubbleSort(Edge edges[], int m) {
    for (int i = 0; i < m - 1; i++) {
        for (int j = 0; j < m - i - 1; j++) {
            if (edges[j].w > edges[j + 1].w) {
                Edge temp = edges[j];
                edges[j] = edges[j + 1];
                edges[j + 1] = temp;
            }
        }
    }
}

void Union(int a[], int n, int u, int v) {
    int temp = a[u];
    for (int i = 0; i < n; i++) {
        if (a[i] == temp)
            a[i] = a[v];
    }
}

int Find(int a[], int n, int u, int v) {
    if (a[u] == a[v])
        return true;
    else
        return false;
}


int main() {
    int n, m;
    cout << "Enter number of vertices and edges: ";
    cin >> n >> m;

    Edge edges[100];

    cout << "Enter edges as: u v w\n";
    for (int i = 0; i < m; i++) {
        cin >> edges[i].u >> edges[i].v >> edges[i].w;
    }

    int parent[100];
    for (int i = 0; i < n; i++)
        parent[i] = i;


    bubbleSort(edges, m);

    int mstWeight = 0;

    cout << "\nEdges in MST:\n";
    for (int i = 0; i < m; i++) {
        int u = edges[i].u;
        int v = edges[i].v;
        int w = edges[i].w;

        if (!Find(parent, n, u, v)) {
            cout << u << " -- " << v << " == " << w << "\n";
            mstWeight += w;
            Union(parent, n, u, v);
        }
    }

    cout << "Total Weight of MST = " << mstWeight << endl;

    return 0;
}
