// ================= SLIDING WINDOW Z-SCORE =================
struct SlidingZ{
    int W; deque q;
    double sum=0, sumsq=0;
    SlidingZ(int w):W(w){}
    pair push(double x,double k){
        q.push_back(x); sum+=x; sumsq+=x*x;
        if(q.size()>W){
            double o=q.front(); q.pop_front();
            sum-=o; sumsq-=o*o;
        }
        if(q.size()<10) return {false,0};
        double mean=sum/q.size();
        double var=sumsq/q.size()-mean*mean;
        double sd=sqrt(max(0.0,var));
        double z = sd>0 ? (x-mean)/sd : 0;
        return {z>k,z};
    }
};

// ================= SKIP LIST NODE =================
struct SkipNode{
    long long key; string value;
    vector next;
    SkipNode(long long k,string v,int lvl)
        :key(k),value(v),next(lvl,nullptr){}
};
