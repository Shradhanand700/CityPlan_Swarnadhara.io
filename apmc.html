Business Plan

The APMC vertical runs digital mandis with transparent auctions, warehousing, and payment rails. It earns from per‑lot platform fees, storage subscriptions, logistics brokerage, and analytics subscriptions for cooperatives and policymakers.

The roadmap starts with high‑volume crops and mandis, rolls out to smaller markets, and later integrates export terminals and cold‑chain operators for premium produce flows.

SDG 2 – Zero Hunger SDG 8 – Decent Work & Growth SDG 9 – Innovation & Infrastructure SDG 12 – Responsible Consumption

Algorithm & DSA View

Hierarchical Market Graph

Model village centres, mandis, warehouses, and ports as a layered graph for produce, bids, and information.

Greedy Auction Clearing

Sort bids with merge sort and clear from highest to lowest while enforcing MSP and quality constraints.

DP Sell vs Store

Dynamic programming compares selling now against storing with decay and forecast prices for future slots.

#include <bits/stdc++.h>
using namespace std;

struct Lot{int id;string commodity;double qty,basePrice;int perishDays;};
struct Bid{int lotId,traderId;double price,qty;};

bool cmpBid(const Bid&a,const Bid&b){
    if(a.price!=b.price)return a.price>b.price;
    return a.traderId<b.traderId;
}

void mergeVec(vector<Bid>&a,int l,int m,int r){
    int n1=m-l+1,n2=r-m;
    vector<Bid>L(n1),R(n2);
    for(int i=0;i<n1;i++)L[i]=a[l+i];
    for(int j=0;j<n2;j++)R[j]=a[m+1+j];
    int i=0,j=0,k=l;
    while(i<n1&&j<n2){
        if(cmpBid(L[i],R[j]))a[k++]=L[i++];
        else a[k++]=R[j++];
    }
    while(i<n1)a[k++]=L[i++];
    while(j<n2)a[k++]=R[j++];
}

void mergeSort(vector<Bid>&a,int l,int r){
    if(l>=r)return;
    int m=l+(r-l)/2;
    mergeSort(a,l,m);
    mergeSort(a,m+1,r);
    mergeVec(a,l,m,r);
}

double clearAuctionGreedy(const Lot&lot,const vector<Bid>&bids){
    double rem=lot.qty,rev=0.0;
    for(auto &b:bids){
        if(b.lotId!=lot.id)continue;
        if(b.price<lot.basePrice)continue;
        if(rem<=0)break;
        double t=min(rem,b.qty);
        rev+=t*b.price; rem-=t;
    }
    return rev;
}

map<pair<int,int>,double> memoAPMC;
double bestRevenueDP(const Lot&lot,int day,int maxDay,
                     const vector<double>&f){
    if(day>=maxDay||lot.perishDays-day<=0)return 0.0;
    auto key=make_pair(lot.id,day);
    if(memoAPMC.count(key))return memoAPMC[key];
    double sellNow=lot.qty*f[day];
    double storedQty=lot.qty*0.98;
    double waitVal=storedQty*f[day+1];
    return memoAPMC[key]=max(sellNow,waitVal);
}

int main(){
    Lot tomato{1,"Tomato",1000,12.0,3};
    vector<Bid>bids={{1,201,13.5,400},{1,202,13.0,300},{1,203,12.5,600}};
    mergeSort(bids,0,(int)bids.size()-1);
    cout<<"Greedy revenue: "
        <<clearAuctionGreedy(tomato,bids)<<"\n";
    vector<double>forecast={12.0,13.0,11.0};
    cout<<"DP best revenue: "
        <<bestRevenueDP(tomato,0,2,forecast)<<"\n";
    return 0;
}
