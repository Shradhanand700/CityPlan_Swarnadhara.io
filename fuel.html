#include <bits/stdc++.h>
using namespace std;

struct DemandPoint{
    int id;
    double x,y,demand;
};
struct Station{
    int id;
    double x,y,capacity;
};

inline double dist2d(double x1,double y1,double x2,double y2){
    const double dx=x1-x2,dy=y1-y2;
    return sqrt(dx*dx+dy*dy);
}

void greedyAssignFuel(const vector<DemandPoint>&dem,vector<Station>&st){
    for(const auto &d:dem){
        int bestId=-1;
        double best=numeric_limits<double>::infinity();
        for(const auto &s:st){
            if(s.capacity<d.demand) continue;
            double c=dist2d(d.x,d.y,s.x,s.y);
            if(c<best){
                best=c;
                bestId=s.id;
            }
        }
        if(bestId!=-1){
            cout<<"Demand "<<d.id<<" -> station "
                <<bestId<<" ("<<best<<" km)\n";
        }else{
            cout<<"No feasible station for demand "<<d.id<<"\n";
        }
    }
}

unordered_map<long long,double> memoFuel;

inline long long keyFuel(int sid,int day){
    return (static_cast<long long>(sid)<<32) | static_cast<unsigned int>(day);
}

double fuelCostDP(int sid,int day,int H,double base){
    if(day>=H) return 0.0;
    const long long key=keyFuel(sid,day);
    auto it=memoFuel.find(key);
    if(it!=memoFuel.end()) return it->second;
    const double today=base*(1.0+0.03*day);
    const double rest=fuelCostDP(sid,day+1,H,base);
    double val=today+rest;
    memoFuel.emplace(key,val);
    return val;
}

int main(){
    vector<DemandPoint> dem;
    dem.reserve(2);
    dem.push_back({1,2,3,500});
    dem.push_back({2,10,8,700});

    vector<Station> st;
    st.reserve(2);
    st.push_back({1,1,2,1000});
    st.push_back({2,9,9,800});

    sort(st.begin(),st.end(),
         [](const Station &a,const Station &b){
             return a.capacity>b.capacity;
         });

    greedyAssignFuel(dem,st);
    cout<<"7-day base cost s1: "
        <<fuelCostDP(1,0,7,1.0)<<"\n";
    return 0;
}
