#include <bits/stdc++.h>
using namespace std;

struct Edge{
    int to;
    double baseTime,congestion;
};
using Graph=vector<vector<Edge>>;

vector<double> shortestPath(const Graph&g,int src){
    const int n=(int)g.size();
    const double INF=numeric_limits<double>::infinity();
    vector<double> dist(n,INF);
    dist[src]=0.0;
    using P=pair<double,int>;
    priority_queue<P,vector<P>,greater<P>> pq;
    pq.emplace(0.0,src);

    while(!pq.empty()){
        const auto [d,u]=pq.top();
        pq.pop();
        if(d>dist[u]) continue;
        for(const auto &e:g[u]){
            const double w=e.baseTime*e.congestion;
            const double nd=dist[u]+w;
            if(nd<dist[e.to]){
                dist[e.to]=nd;
                pq.emplace(nd,e.to);
            }
        }
    }
    return dist;
}

struct IntersectionState{
    int id;
    vector<int> queueLengths;
};

int chooseGreenPhaseGreedy(const IntersectionState&st){
    int best=0;
    int bestQ=-1;
    const int n=(int)st.queueLengths.size();
    for(int i=0;i<n;i++){
        if(st.queueLengths[i]>bestQ){
            bestQ=st.queueLengths[i];
            best=i;
        }
    }
    return best;
}

unordered_map<int,double> memoTraffic;

double corridorCostDP(int cid,int slot,int maxSlot,const vector<double>&delay){
    if(slot>=maxSlot) return 0.0;
    const int key=(cid<<16)|slot;
    auto it=memoTraffic.find(key);
    if(it!=memoTraffic.end()) return it->second;
    double val=delay[slot]+corridorCostDP(cid,slot+1,maxSlot,delay);
    memoTraffic.emplace(key,val);
    return val;
}

int main(){
    Graph g(4);
    g[0].push_back({1,10.0,1.2});
    g[1].push_back({2, 8.0,1.5});
    g[2].push_back({3, 6.0,1.1});

    auto dist=shortestPath(g,0);
    cout<<"0->3 time: "<<dist[3]<<"\n";

    IntersectionState st{1,{10,25,5,12}};
    cout<<"Chosen phase: "
        <<chooseGreenPhaseGreedy(st)<<"\n";

    vector<double> delay={100,120,130,110};
    cout<<"Corridor delay: "
        <<corridorCostDP(0,0,(int)delay.size(),delay)<<"\n";
    return 0;
}
