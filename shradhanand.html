Module 1 ‚Äì Health Care System

Emergency Routing & Hospital Beds Resource Manager

Instead of hardcoding city roads, hospitals, and patients, the system uses a JSON input file. This makes the system scalable and realistic, allowing city data to change without modifying the algorithm.

What This JSON File Is Used For

  • Defines the city road network with distances
  • Stores ambulance locations and readiness
  • Stores patient location and severity
  • Stores hospital locations and available beds

Sample JSON Input File

{
  "roads": [
    { "from": "A", "to": "B", "distance": 3.2 },
    { "from": "B", "to": "C", "distance": 1.5 },
    { "from": "C", "to": "D", "distance": 2.8 },
    { "from": "A", "to": "D", "distance": 6.5 }
  ],
  "ambulances": [
    { "id": "AMB01", "location": "A", "ready": true },
    { "id": "AMB02", "location": "B", "ready": true }

    { "id": "PAT101", "location": "C", "severity": 4 }

  "hospitals": [
    { "id": "HOSP1", "location": "D", "beds_free": 5 }
  ]
}

How the Algorithm Uses the JSON File (Approach-1)

  • JSON file is read at runtime
  • Locations converted to graph nodes
  • Roads become weighted edges
  • Dijkstra runs from patient location
  • Nearest hospital with free beds selected

Integrated C++ Code (JSON-Driven Emergency Routing)

// File: healthcare.cpp
// Compile: g++ -std=c++17 healthcare.cpp -O2 -o healthcare
// Run: ./healthcare

#include <bits/stdc++.h>
#include "json.hpp"
using namespace std;
using json = nlohmann::json;

const double INF = 1e18;

struct Edge {
    int to;
    double w;
};

struct Graph {
    int n;
    vector<vector<Edge>> adj;

    Graph(int n = 0) : n(n), adj(n) {}

    void addEdge(int u, int v, double w) {
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    vector<double> dijkstra(int src) {
        vector<double> dist(n, INF);
        priority_queue<pair<double,int>,
            vector<pair<double,int>>, greater<>> pq;

        dist[src] = 0;
        pq.push({0, src});

        while (!pq.empty()) {
            auto [d,u] = pq.top(); pq.pop();
            if (d > dist[u]) continue;

            for (auto &e : adj[u]) {
                if (dist[e.to] > d + e.w) {
                    dist[e.to] = d + e.w;
                    pq.push({dist[e.to], e.to});
                }
            }
        }
        return dist;
    }
};

int main() {
    ifstream file("healthcare_input.json");
    json data; file >> data;

    unordered_map<string,int> id;
    int idx = 0;

    auto getId = [&](const string &s) {
        if (!id.count(s)) id[s] = idx++;
        return id[s];
    };

    for (auto &r : data["roads"]) {
        getId(r["from"]);
        getId(r["to"]);
    }

    Graph g(idx);
    for (auto &r : data["roads"]) {
        g.addEdge(getId(r["from"]), getId(r["to"]), r["distance"]);
    }

    int patientNode = getId(data["patients"][0]["location"]);
    auto dist = g.dijkstra(patientNode);

    double best = INF;
    string bestHospital = "NONE";

    for (auto &h : data["hospitals"]) {
        if (h["beds_free"].get<int>() > 0) {
            int hn = getId(h["location"]);
            if (dist[hn] < best) {
                best = dist[hn];
                bestHospital = h["id"];
            }
        }
    }

    cout << "Nearest Hospital: " << bestHospital << "\\n";
    cout << "Distance: " << best << " km\\n";
}
