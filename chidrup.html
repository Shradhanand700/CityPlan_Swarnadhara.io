Fuel Station Management – Logic

Graph & Partitioning

City zones (coastal, core, highway) are subgraphs; each has demand nodes and candidate stations joined by weighted road edges.

Shortest Path Trees

Shortest path trees provide travel times used to evaluate which station is most convenient for each demand cluster.

Greedy + DP

Greedy assigns demand to the nearest feasible station; DP accumulates multi-day logistics and stock costs.

APMC Management – Logic

Hierarchical Market Graph

Village collection, mandis, warehouses, and port terminals form levels in a graph for flow of produce and bids.

Greedy Auctions

Sorted bids are cleared greedily from highest to lowest while enforcing MSP and reserve price constraints.

DP Sell vs Store

DP compares sell-today revenue with discounted future revenue after storage losses and price forecasts.

City Traffic Management – Logic

Road Network Graph

Intersections are nodes; edges carry base travel time and congestion factors and feed into shortest-path routing.

Greedy Signal Control

Local controllers pick the movement with the highest queue length subject to safety and minimum-green rules.

DP Over Time Slots

DP accumulates corridor delay values over time slots and evaluates strategies across peak and off-peak windows.

Construction Contracting – Logic

Project & Logistics Graphs

Task dependencies form project graphs; depots, sites, and materials are connected in logistics graphs for routing.

Greedy Scheduling

Tasks are sorted by deadlines and penalties to minimise total delay cost under crew and resource limits.

DP Cost/Risk

DP sums base cost and dynamic risk multipliers over days, capturing monsoon or supply uncertainty.

Fuel Station Planning – C++ Sketch


#include <bits/stdc++.h>
using namespace std;

struct DemandPoint{
    int id; double x,y,demand;
};
struct Station{
    int id; double x,y,capacity;
};
double dist2d(double x1,double y1,double x2,double y2){
    double dx=x1-x2,dy=y1-y2;
    return sqrt(dx*dx+dy*dy);
}
void greedyAssignFuel(vector<DemandPoint>&dem,vector<Station>&st){
    for(auto &d:dem){
        int bestId=-1; double best=1e18;
        for(auto &s:st){
            if(s.capacity<d.demand)continue;
            double c=dist2d(d.x,d.y,s.x,s.y);
            if(c<best){best=c;bestId=s.id;}
        }
        if(bestId!=-1)
            cout<<"Fuel demand "<<d.id<<" -> station "
                <<bestId<<" ("<<best<<" km)\n";
        else
            cout<<"No feasible station for demand "<<d.id<<"\n";
    }
}
map<pair<int,int>,double> memoFuel;
double fuelCostDP(int sid,int day,int H,double base){
    if(day>=H)return 0.0;
    auto key=make_pair(sid,day);
    if(memoFuel.count(key))return memoFuel[key];
    double today=base*(1.0+0.03*day);
    return memoFuel[key]=today+fuelCostDP(sid,day+1,H,base);
}
int main(){
    vector<DemandPoint> dem={{1,2,3,500},{2,10,8,700}};
    vector<Station> st={{1,1,2,1000},{2,9,9,800}};
    sort(st.begin(),st.end(),
         [](auto &a,auto &b){return a.capacity>b.capacity;});
    greedyAssignFuel(dem,st);
    cout<<"Fuel 7-day base cost s1: "
        <<fuelCostDP(1,0,7,1.0)<<"\n";
    return 0;
}

APMC Management – C++ Sketch


#include <bits/stdc++.h>
using namespace std;

struct Lot{int id;string commodity;double qty,basePrice;int perishDays;};
struct Bid{int lotId,traderId;double price,qty;};
bool cmpBid(const Bid&a,const Bid&b){
    if(a.price!=b.price)return a.price>b.price;
    return a.traderId<b.traderId;
}
void mergeVec(vector<Bid>&a,int l,int m,int r){
    int n1=m-l+1,n2=r-m;
    vector<Bid>L(n1),R(n2);
    for(int i=0;i<n1;i++)L[i]=a[l+i];
    for(int j=0;j<n2;j++)R[j]=a[m+1+j];
    int i=0,j=0,k=l;
    while(i<n1&&j<n2){
        if(cmpBid(L[i],R[j]))a[k++]=L[i++];
        else a[k++]=R[j++];
    }
    while(i<n1)a[k++]=L[i++];
    while(j<n2)a[k++]=R[j++];
}
void mergeSort(vector<Bid>&a,int l,int r){
    if(l>=r)return;
    int m=l+(r-l)/2;
    mergeSort(a,l,m);
    mergeSort(a,m+1,r);
    mergeVec(a,l,m,r);
}
double clearAuctionGreedy(const Lot&lot,const vector<Bid>&bids){
    double rem=lot.qty,rev=0.0;
    for(auto &b:bids){
        if(b.lotId!=lot.id)continue;
        if(b.price<lot.basePrice)continue;
        if(rem<=0)break;
        double t=min(rem,b.qty);
        rev+=t*b.price; rem-=t;
    }
    return rev;
}
map<pair<int,int>,double> memoAPMC;
double bestRevenueDP(const Lot&lot,int day,int maxDay,
                     const vector<double>&f){
    if(day>=maxDay||lot.perishDays-day<=0)return 0.0;
    auto key=make_pair(lot.id,day);
    if(memoAPMC.count(key))return memoAPMC[key];
    double sellNow=lot.qty*f[day];
    double storedQty=lot.qty*0.98;
    double waitVal=storedQty*f[day+1];
    return memoAPMC[key]=max(sellNow,waitVal);
}
int main(){
    Lot tomato{1,"Tomato",1000,12.0,3};
    vector<Bid>bids={{1,201,13.5,400},{1,202,13.0,300},{1,203,12.5,600}};
    mergeSort(bids,0,(int)bids.size()-1);
    cout<<"APMC greedy revenue: "
        <<clearAuctionGreedy(tomato,bids)<<"\n";
    vector<double>forecast={12.0,13.0,11.0};
    cout<<"APMC DP best revenue: "
        <<bestRevenueDP(tomato,0,2,forecast)<<"\n";
    return 0;
}

Traffic Management – C++ Sketch


#include <bits/stdc++.h>
using namespace std;

struct Edge{int to;double baseTime,congestion;};
using Graph=vector<vector<Edge>>;
vector<double> shortestPath(const Graph&g,int src){
    int n=g.size(); const double INF=1e18;
    vector<double>dist(n,INF); dist[src]=0;
    using P=pair<double,int>;
    priority_queue<P,vector<P>,greater<P>>pq;
    pq.push({0,src});
    while(!pq.empty()){
        auto[d,u]=pq.top();pq.pop();
        if(d>dist[u])continue;
        for(auto &e:g[u]){
            double w=e.baseTime*e.congestion;
            if(dist[u]+w<dist[e.to]){
                dist[e.to]=dist[u]+w;
                pq.push({dist[e.to],e.to});
            }
        }
    }
    return dist;
}
struct IntersectionState{int id;vector<int>queueLengths;};
int chooseGreenPhaseGreedy(const IntersectionState&st){
    int best=0,bq=-1;
    for(int i=0;i<(int)st.queueLengths.size();i++)
        if(st.queueLengths[i]>bq){bq=st.queueLengths[i];best=i;}
    return best;
}
map<pair<int,int>,double> memoTraffic;
double corridorCostDP(int cid,int slot,int maxSlot,const vector<double>&delay){
    if(slot>=maxSlot)return 0.0;
    auto key=make_pair(cid,slot);
    if(memoTraffic.count(key))return memoTraffic[key];
    return memoTraffic[key]=delay[slot]+
        corridorCostDP(cid,slot+1,maxSlot,delay);
}
int main(){
    Graph g(4);
    g[0].push_back({1,10.0,1.2});
    g[1].push_back({2,8.0,1.5});
    g[2].push_back({3,6.0,1.1});
    auto dist=shortestPath(g,0);
    cout<<"Traffic 0->3 time: "<<dist[3]<<"\n";
    IntersectionState st{1,{10,25,5,12}};
    cout<<"Chosen phase: "<<chooseGreenPhaseGreedy(st)<<"\n";
    vector<double>delay={100,120,130,110};
    cout<<"Corridor delay: "
        <<corridorCostDP(0,0,(int)delay.size(),delay)<<"\n";
    return 0;
}

Construction Contracting – C++ Sketch


#include <bits/stdc++.h>
using namespace std;

struct Task{int id;string name;int duration,deadline;double penalty;};
bool taskCmp(const Task&a,const Task&b){
    if(a.deadline!=b.deadline)return a.deadline<b.deadline;
    return a.penalty>b.penalty;
}
struct LEdge{int to;double cost;};
using LGraph=vector<vector<LEdge>>;
vector<double> shortestPathL(const LGraph&g,int src){
    int n=g.size(); const double INF=1e18;
    vector<double>dist(n,INF); dist[src]=0;
    using P=pair<double,int>;
    priority_queue<P,vector<P>,greater<P>>pq;
    pq.push({0,src});
    while(!pq.empty()){
        auto[d,u]=pq.top();pq.pop();
        if(d>dist[u])continue;
        for(auto &e:g[u]){
            if(dist[u]+e.cost<dist[e.to]){
                dist[e.to]=dist[u]+e.cost;
                pq.push({dist[e.to],e.to});
            }
        }
    }
    return dist;
}
map<pair<int,int>,double> memoConstr;
double projectCostDP(int pid,int day,int H,double base,double risk){
    if(day>=H)return 0.0;
    auto key=make_pair(pid,day);
    if(memoConstr.count(key))return memoConstr[key];
    double today=base*(1.0+risk*day);
    return memoConstr[key]=today+projectCostDP(pid,day+1,H,base,risk);
}
int main(){
    vector<Task>tasks={{1,"Foundations",30,60,5000},
                        {2,"Piers",40,120,4000},
                        {3,"Deck",50,180,6000}};
    sort(tasks.begin(),tasks.end(),taskCmp);
    int cur=0; double totalPenalty=0;
    for(auto &t:tasks){
        int finish=cur+t.duration;
        int delay=max(0,finish-t.deadline);
        double pen=delay*t.penalty;
        totalPenalty+=pen;
        cout<<"Task "<<t.id<<" delay="<<delay
            <<" penalty="<<pen<<"\n";
        cur=finish;
    }
    cout<<"Total penalty: "<<totalPenalty<<"\n";
    LGraph g(3);
    g[0].push_back({1,10.0}); g[0].push_back({2,20.0});
    auto d=shortestPathL(g,0);
    cout<<"Depot->Site1 cost: "<<d[1]<<"\n";
    cout<<"10-day project cost: "
        <<projectCostDP(1,0,10,100000,0.02)<<"\n";
    return 0;
}
