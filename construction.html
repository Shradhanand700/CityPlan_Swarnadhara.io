Project DAGs

Model tasks and dependencies as directed acyclic graphs for critical path and sequencing analysis.

Greedy Scheduling

Logistics & DP

Use shortest paths for routing materials and DP to accumulate cost and risk over a planning horizon.

#include <bits/stdc++.h>
using namespace std;

struct Task{
    int id;
    string name;
    int duration,deadline;
    double penalty;
};

inline bool taskCmp(const Task&a,const Task&b){
    if(a.deadline!=b.deadline) return a.deadline<b.deadline;
    return a.penalty>b.penalty;
}

struct LEdge{
    int to;
    double cost;
};
using LGraph=vector<vector<LEdge>>;

vector<double> shortestPathL(const LGraph&g,int src){
    const int n=(int)g.size();
    const double INF=numeric_limits<double>::infinity();
    vector<double> dist(n,INF);
    dist[src]=0.0;
    using P=pair<double,int>;
    priority_queue<P,vector<P>,greater<P>> pq;
    pq.emplace(0.0,src);

    while(!pq.empty()){
        const auto [d,u]=pq.top();
        pq.pop();
        if(d>dist[u]) continue;
        for(const auto &e:g[u]){
            const double nd=dist[u]+e.cost;
            if(nd<dist[e.to]){
                dist[e.to]=nd;
                pq.emplace(nd,e.to);
            }
        }
    }
    return dist;
}

unordered_map<int,double> memoConstr;

double projectCostDP(int pid,int day,int H,double base,double risk){
    if(day>=H) return 0.0;
    const int key=(pid<<16)|day;
    auto it=memoConstr.find(key);
    if(it!=memoConstr.end()) return it->second;
    const double today=base*(1.0+risk*day);
    double val=today+projectCostDP(pid,day+1,H,base,risk);
    memoConstr.emplace(key,val);
    return val;
}

int main(){
    vector<Task> tasks;
    tasks.reserve(3);
    tasks.push_back({1,"Foundations",30,60,5000});
    tasks.push_back({2,"Piers",40,120,4000});
    tasks.push_back({3,"Deck",50,180,6000});

    sort(tasks.begin(),tasks.end(),taskCmp);

    int cur=0;
    double totalPenalty=0.0;
    for(const auto &t:tasks){
        const int finish=cur+t.duration;
        const int delay=max(0,finish-t.deadline);
        const double pen=delay*t.penalty;
        totalPenalty+=pen;
        cout<<"Task "<<t.id<<" delay="<<delay
            <<" penalty="<<pen<<"\n";
        cur=finish;
    }
    cout<<"Total penalty: "<<totalPenalty<<"\n";

    LGraph g(3);
    g[0].push_back({1,10.0});
    g[0].push_back({2,20.0});
    auto d=shortestPathL(g,0);
    cout<<"Depot->Site1 cost: "<<d[1]<<"\n";
    cout<<"10-day project cost: "
        <<projectCostDP(1,0,10,100000,0.02)<<"\n";
    return 0;
}
