Module 1 ‚Äì Health Care System

A. Emergency Routing & Dispatch

Computes the fastest ambulance ‚Üí patient ‚Üí hospital route under dynamic conditions using shortest path algorithms.

How the Code Works

The city map is modeled as a graph. Dijkstra computes shortest paths from the patient to all nodes. The nearest hospital with available beds is selected, and the best ambulance is chosen based on distance and severity.

Efficiency Analysis

AlgorithmTimeSpace
DijkstraO(E log V)O(V + E)
A*O(E log V)O(V)
Bellman-FordO(VE)O(V)

B. Hospital Bed & Resource Manager

Maintains hospital bed occupancy and schedules surgeries efficiently.

How the Code Works

A segment tree supports fast range updates for bed occupancy. A Fenwick tree tracks total admissions efficiently.

// ===== GRAPH & ROUTING =====
#include <bits/stdc++.h>
using namespace std;
const double INF = 1e18;

struct Edge{ int to; double w; };

struct Graph{
    int n;
    vector<vector<Edge>> adj;

    Graph(int n):n(n),adj(n){}

    void add(int u,int v,double w,bool und=true){
        adj[u].push_back({v,w});
        if(und) adj[v].push_back({u,w});
    }

    vector<double> dijkstra(int s){
        vector<double> dist(n,INF);
        priority_queue<pair<double,int>, vector<pair<double,int>>, greater<>> pq;
        dist[s]=0; pq.push({0,s});
        while(!pq.empty()){
            auto [d,u]=pq.top(); pq.pop();
            if(d>dist[u]) continue;
            for(auto &e:adj[u]){
                if(dist[e.to] > d + e.w){
                    dist[e.to] = d + e.w;
                    pq.push({dist[e.to], e.to});
                }
            }
        }
        return dist;
    }
};

// ===== BED MANAGEMENT =====
struct Fenwick{
    int n;
    vector<int> bit;
    Fenwick(int n):n(n),bit(n+1,0){}
    void add(int i,int v){
        for(++i;i<=n;i+=i&-i) bit[i]+=v;
    }
    int sum(int i){
        int r=0;
        for(++i;i>0;i-=i&-i) r+=bit[i];
        return r;
    }
};
