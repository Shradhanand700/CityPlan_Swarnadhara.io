// ===== DSU =====
#include <bits/stdc++.h>
using namespace std;

struct DSU{
    vector<int> p,r;
    DSU(int n):p(n),r(n,0){ iota(p.begin(),p.end(),0); }
    int find(int x){
        return p[x]==x?x:p[x]=find(p[x]);
    }
    bool unite(int a,int b){
        a=find(a); b=find(b);
        if(a==b) return false;
        if(r[a]<r[b]) swap(a,b);
        p[b]=a;
        if(r[a]==r[b]) r[a]++;
        return true;
    }
};

// ===== TRIE =====
struct Trie{
    struct Node{
        unordered_map<char,int> ch;
        bool end=false;
    };
    vector<Node> t;
    Trie(){ t.emplace_back(); }
    void insert(string s){
        int cur=0;
        for(char &c:s){
            c=tolower(c);
            if(!t[cur].ch[c]){
                t[cur].ch[c]=t.size();
                t.emplace_back();
            }
            cur=t[cur].ch[c];
        }
        t[cur].end=true;
    }
};

// ===== BLOOM FILTER =====
struct Bloom{
    vector<bool> b;
    int k;
    Bloom(int m,int k):b(m,false),k(k){}
    void add(string s){
        hash<string> h;
        for(int i=0;i<k;i++)
            b[(h(s)+i*i)%b.size()] = true;
    }
    bool maybe(string s){
        hash<string> h;
        for(int i=0;i<k;i++)
            if(!b[(h(s)+i*i)%b.size()])
                return false;
        return true;
    }
};
