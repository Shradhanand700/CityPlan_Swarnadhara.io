A. Transaction Stream Monitoring & Fraud Detection

Processes live banking transactions to detect abnormal behavior, duplicate patterns, and suspicious account connectivity.

How the Code Works

A sliding-window z-score detects unusually large transactions. Bloom filters reject duplicate transaction IDs efficiently. Graph traversal (BFS/DFS) explores suspicious money-flow networks.

C. Ledger Management & Fast Lookup

Maintains ordered transaction history with ultra-fast lookup capabilities.

How the Code Works

Skip Lists maintain ordered ledger entries with probabilistic guarantees. std::map ensures strict log-time ordering. unordered_map allows constant-time transaction ID lookups.

// ================= SLIDING WINDOW Z-SCORE =================
struct SlidingZ{
    int W; deque<double> q;
    double sum=0, sumsq=0;
    SlidingZ(int w):W(w){}
    pair<bool,double> push(double x,double k){
        q.push_back(x); sum+=x; sumsq+=x*x;
        if(q.size()>W){
            double o=q.front(); q.pop_front();
            sum-=o; sumsq-=o*o;
        }
        if(q.size()<10) return {false,0};
        double mean=sum/q.size();
        double var=sumsq/q.size()-mean*mean;
        double sd=sqrt(max(0.0,var));
        double z = sd>0 ? (x-mean)/sd : 0;
        return {z>k,z};
    }
};

// ================= SKIP LIST NODE =================
struct SkipNode{
    long long key; string value;
    vector<SkipNode*> next;
    SkipNode(long long k,string v,int lvl)
        :key(k),value(v),next(lvl,nullptr){}
};
