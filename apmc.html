Algorithm & DSA View

Hierarchical Market Graph

Model village centres, mandis, warehouses, and ports as a layered network for produce, bids, and information.

Greedy Auction Clearing

Sort bids with merge sort and clear lots from highest bid down, under MSP and quality constraints.

DP Sell vs Store

Dynamic programming compares selling now versus storing with loss and future price forecasts.

#include <bits/stdc++.h>
using namespace std;

struct Lot{
    int id;
    string commodity;
    double qty,basePrice;
    int perishDays;
};
struct Bid{
    int lotId,traderId;
    double price,qty;
};

inline bool cmpBid(const Bid&a,const Bid&b){
    if(a.price!=b.price) return a.price>b.price;
    return a.traderId<b.traderId;
}

void mergeVec(vector<Bid>&a,int l,int m,int r){
    const int n1=m-l+1,n2=r-m;
    vector<Bid> L; L.reserve(n1);
    vector<Bid> R; R.reserve(n2);
    for(int i=0;i<n1;i++) L.push_back(a[l+i]);
    for(int j=0;j<n2;j++) R.push_back(a[m+1+j]);
    int i=0,j=0,k=l;
    while(i<n1 && j<n2){
        if(cmpBid(L[i],R[j])) a[k++]=L[i++];
        else a[k++]=R[j++];
    }
    while(i<n1) a[k++]=L[i++];
    while(j<n2) a[k++]=R[j++];
}

void mergeSort(vector<Bid>&a,int l,int r){
    if(l>=r) return;
    int m=l+(r-l)/2;
    mergeSort(a,l,m);
    mergeSort(a,m+1,r);
    mergeVec(a,l,m,r);
}

double clearAuctionGreedy(const Lot&lot,const vector<Bid>&bids){
    double rem=lot.qty;
    double revenue=0.0;
    for(const auto &b:bids){
        if(b.lotId!=lot.id || b.price<lot.basePrice) continue;
        if(rem<=0) break;
        const double traded=min(rem,b.qty);
        revenue+=traded*b.price;
        rem-=traded;
    }
    return revenue;
}

unordered_map<long long,double> memoAPMC;
inline long long keyAP(int lotId,int day){
    return (static_cast<long long>(lotId)<<32) | static_cast<unsigned int>(day);
}

double bestRevenueDP(const Lot&lot,int day,int maxDay,
                     const vector<double>&forecast){
    if(day>=maxDay || lot.perishDays-day<=0) return 0.0;
    const long long key=keyAP(lot.id,day);
    auto it=memoAPMC.find(key);
    if(it!=memoAPMC.end()) return it->second;

    const double sellNow=lot.qty*forecast[day];
    const double storedQty=lot.qty*0.98; // 2% loss
    const double waitVal=storedQty*forecast[day+1];
    double val=max(sellNow,waitVal);
    memoAPMC.emplace(key,val);
    return val;
}

int main(){
    Lot tomato{1,"Tomato",1000,12.0,3};
    vector<Bid> bids;
    bids.reserve(3);
    bids.push_back({1,201,13.5,400});
    bids.push_back({1,202,13.0,300});
    bids.push_back({1,203,12.5,600});

    mergeSort(bids,0,(int)bids.size()-1);

    cout<<"Greedy revenue: "
        <<clearAuctionGreedy(tomato,bids)<<"\n";

    vector<double> forecast={12.0,13.0,11.0};
    cout<<"DP best revenue: "
        <<bestRevenueDP(tomato,0,2,forecast)<<"\n";
    return 0;
}
