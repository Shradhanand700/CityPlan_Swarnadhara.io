class MaxHeap {
    int arr[100];
    int size;

public:
    MaxHeap() {
        size = 0;
    }

    // Insert element into heap
    void insert(int val) {
        size++;
        int i = size;

        // Move up until heap property is satisfied
        while (i > 1 && arr[i / 2] < val) {
            arr[i] = arr[i / 2];
            i = i / 2;
        }
        arr[i] = val;
    }

    // Delete maximum element (root)
    void deleteMax() {
        if (size == 0) {
            cout << "Heap is empty\n";
            return;
        }

        int last = arr[size];
        size--;

        int i = 1;
        int child;

        while (2 * i <= size) {
            child = 2 * i;

            if (child + 1 <= size && arr[child + 1] > arr[child])
                child++;

            if (last >= arr[child])
                break;

            arr[i] = arr[child];
            i = child;
        }
        arr[i] = last;
    }

    // Display heap
    void display() {
        for (int i = 1; i <= size; i++)
            cout << arr[i] << " ";
        cout << endl;
    }
};

int main() {
    MaxHeap h;

    h.insert(10);
    h.insert(40);
    h.insert(30);
    h.insert(60);
    h.insert(20);

    cout << "Heap elements: ";
    h.display();

    h.deleteMax();

    cout << "After deleting max: ";
    h.display();

    return 0;
}




    struct Node {
    int data;
    Node* left;
    Node* right;
};

Node* insert(Node* root, int key) {
    if (!root) return new Node{key, NULL, NULL};
    if (key < root->data)
        root->left = insert(root->left, key);
    else
        root->right = insert(root->right, key);
    return root;
}



    void dfs(int v, vector adj[], vector& visited) {
    visited[v] = true;
    for (int u : adj[v]) {
        if (!visited[u])
            dfs(u, adj, visited);
    }
}

    void dijkstra(int src, vector> adj[]) {
    priority_queue, vector>, greater<>> pq;
    vector dist(100, INT_MAX);
    dist[src] = 0;
    pq.push({0, src});

    while (!pq.empty()) {
        int u = pq.top().second;
        pq.pop();
        for (auto [v,w] : adj[u]) {
            if (dist[v] > dist[u] + w) {
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }
}

    void build(int idx, int l, int r) {
    if (l == r) return;
    int mid = (l+r)/2;
    build(2*idx, l, mid);
    build(2*idx+1, mid+1, r);
}
