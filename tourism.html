Module 3 ‚Äì Tourism & Resort Booking System

A. Booking Engine & Dynamic Pricing

This module manages resort occupancy, processes booking requests based on priority, and ensures maximum utilization while preventing overbooking.

How the Algorithm Works

A segment tree tracks daily occupancy efficiently. Booking requests are sorted by priority so VIP customers are processed first. If capacity allows, bookings are accepted and revenue is updated.

Efficiency Analysis

ComponentTimeSpace
Segment TreeO(log n)O(n)
Priority SortingO(m log m)O(m)

B. Itinerary Planner

Generates an efficient travel plan covering multiple tourist spots using distance-based heuristics.

How the Algorithm Works

Tourist locations are treated as nodes in a graph. A nearest-neighbour heuristic selects the closest unvisited location, minimizing total travel distance.

Efficiency Analysis

AlgorithmTimeSpace
Nearest NeighbourO(n¬≤)O(n)

Integrated C++ Code (JSON Driven)

#include <bits/stdc++.h>
#include "json.hpp"
using namespace std;
using json = nlohmann::json;

struct Seg {
    int n; vector<int> t;
    Seg(int N){ n=1; while(n<N) n<<=1; t.assign(2*n,0); }
    void add(int i,int v){
        i+=n; t[i]+=v;
        for(i>>=1;i;i>>=1)
            t[i]=max(t[2*i],t[2*i+1]);
    }
    int maxq(int l,int r){
        int res=0;
        for(l+=n,r+=n;l<=r;l>>=1,r>>=1){
            if(l&1) res=max(res,t[l++]);
            if(!(r&1)) res=max(res,t[r--]);
        }
        return res;
    }
};

int main(){
    ifstream f("booking_input.json");
    json jb; f>>jb;

    int cap = jb["resort"]["capacity_per_day"];
    int price = jb["resort"]["base_price"];

    struct B{int id,l,r,p;};
    vector<B> bk;
    for(auto &x:jb["bookings"])
        bk.push_back({x["id"],x["start_day"],x["end_day"],x["priority"]});

    sort(bk.begin(),bk.end(),
         [](auto &a,auto &b){return a.p>b.p;});

    Seg seg(365);
    int revenue=0;

    for(auto &b:bk){
        if(seg.maxq(b.l,b.r)<cap){
            for(int d=b.l;d<=b.r;d++) seg.add(d,1);
            revenue+=(b.r-b.l+1)*price;
        }
    }
    cout<<"Total Revenue: "<<revenue<<"\n";
}
